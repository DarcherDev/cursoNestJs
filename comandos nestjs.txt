======================================================================
GUÍA DE COMANDOS NESTJS Y HERRAMIENTAS
======================================================================

--- GESTIÓN DEL PROYECTO ---

# Crear nuevo proyecto
nest new nombre-proyecto

# Iniciar servidor (modo producción)
npm run start

# Iniciar servidor (modo desarrollo - watch mode)
npm run start:dev
# URL por defecto: http://localhost:3000

# Detener servidor
Ctrl + C

# Compilar proyecto (Build para producción)
npm run build

# Levantar proyecto compilado
nest start

# Linting (Buscar errores de estilo/código)
npm run lint

--- GENERADORES (NEST CLI) ---
# Estructura: nest g <tipo> <nombre> [ruta] [opciones]

# Ayuda general
nest g --help

# Elementos comunes
nest g mo users           # Módulo (Module)
nest g co users           # Controlador (Controller)
nest g s users            # Servicio (Service)
nest g res users          # Recurso CRUD completo (Resource)

# Otros elementos
nest g cl nombre users/dto --no-spec  # Clase (útil para DTOs)
nest g gu auth            # Guard
nest g pi validation      # Pipe
nest g mi logger          # Middleware
nest g f http-exception   # Filter
nest g itc logging        # Interceptor

# Opciones útiles
--no-spec    # No genera archivos de pruebas (.spec.ts)
--dry-run    # Simula la creación sin escribir archivos (para probar)
--flat       # No crea una carpeta nueva para el elemento

--- VALIDACIÓN (DTOs) ---

# Instalar dependencias necesarias
npm i class-validator class-transformer

--- DOCUMENTACIÓN (SWAGGER) ---

# Instalar dependencias
npm install --save @nestjs/swagger

--- PRISMA (ORM) ---

# Instalación
npm install prisma --save-dev
npm install @prisma/client

# Inicializar Prisma
npx prisma init

# Crear migración (sincronizar schema.prisma con DB)
npx prisma migrate dev --name nombre_migracion

# Generar cliente (actualizar tipos después de cambios en schema)
npx prisma generate

# Abrir interfaz visual de la base de datos
npx prisma studio

--- DOCKER ---

# Levantar contenedores (en segundo plano)
docker-compose up -d

# Bajar contenedores
docker-compose down

# Bajar contenedores y borrar volúmenes (resetear DB)
docker-compose down -v
